function setChat(chatId, e) {
    if (chatId === chatID) {
        return false;
    }

    $.ajax({
        url: PATH+'chat/?action=setChat&chatid='+chatId.toString(),
        dataType: "JSON",
        cache: false
    }).done(function(data) {
        if (!data.success) {
            return;
        }

        wsUnSubscribe('chat' + chatID);
        wsReconnect();
        var reconnectInterval = setInterval(function() {
            if (!wsIsConnected()) {
                return false;
            }
            clearInterval(reconnectInterval);

            chatID = chatId;
            current_id = parseInt(data.min_id);
            CHAT.chat.html('');
            CHAT.initChatSocket(chatID);
            CHAT.appendMessages(data.messages.reverse() || []);
            CHAT.appendPinMessage(data.pin_message);
            CHAT.container.find('.baron__scroller').scrollTop(CHAT.container.find('.baron__scroller').innerHeight());

            var wrap = $('.sys-chat-langs');
            wrap.find('.chat-langs-item').removeClass('active');
            e.addClass('active');
        }, 100);
    }).fail(function(jqXHR, textStatus) {
        CHAT.chat.html("<center>Ошибка загрузки</center>" + CHAT.chat.html());
    }).always(function(){
        CHAT.sending = false;
        CHAT.container.find(".chat-loader").removeClass("active");
    });
}
